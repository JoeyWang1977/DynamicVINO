38 bool openvino_service::EmotionsDetection::fetchResults() {
39  bool can_fetch = openvino_service::BaseInference::fetchResults();
40  if (!can_fetch) return false;
41  int label_length = static_cast<int>(valid_network_->getLabels().size());
42  std::string output_name = valid_network_->getOutputName();
43  InferenceEngine::Blob::Ptr
44  emotions_blob = getEngine()->getRequest()->GetBlob(output_name);
47  long num_of_channels = emotions_blob->getTensorDesc().getDims().at(1);
48  if (num_of_channels != label_length) {
49  throw std::logic_error("Output size (" + std::to_string(num_of_channels) +
50  ") of the Emotions Recognition network is not equal "
51  "to used emotions vector size (" +
52  std::to_string(label_length )+ ")");
53  }
56  auto emotions_values = emotions_blob->buffer().as<float *>();
57  for (int idx = 0; idx < results_.size(); ++idx) {
58  auto output_idx_pos = emotions_values + idx;
59  long max_prob_emotion_idx =
60  std::max_element(output_idx_pos, output_idx_pos + label_length) -
61  output_idx_pos;
62  results_[idx].label = valid_network_->getLabels()[max_prob_emotion_idx];
63  }
64  return true;
65 };
