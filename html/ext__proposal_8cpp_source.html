31 static
32 void generate_anchors(int base_size, float* ratios,
33  float* scales, const int num_ratios,
34  const int num_scales, float* anchors,
35  float coordinates_offset, bool shift_anchors, bool round_ratios) {
36  // base box's width & height & center location
37  const float base_area = static_cast<float>(base_size * base_size);
38  const float half_base_size = base_size * 0.5f;
39  const float center = 0.5f * (base_size - coordinates_offset);
40 
41  // enumerate all transformed boxes
42  for (int ratio = 0; ratio < num_ratios; ++ratio) {
43  // transformed width & height for given ratio factors
44  float ratio_w;
45  float ratio_h;
46  if (round_ratios) {
47  ratio_w = std::roundf(std::sqrt(base_area / ratios[ratio]));
48  ratio_h = std::roundf(ratio_w * ratios[ratio]);
49  } else {
50  ratio_w = std::sqrt(base_area / ratios[ratio]);
51  ratio_h = ratio_w * ratios[ratio];
52  }
53 
54  float * const p_anchors_wm = anchors + 0 * num_ratios * num_scales + ratio * num_scales;
55  float * const p_anchors_hm = anchors + 1 * num_ratios * num_scales + ratio * num_scales;
56  float * const p_anchors_wp = anchors + 2 * num_ratios * num_scales + ratio * num_scales;
57  float * const p_anchors_hp = anchors + 3 * num_ratios * num_scales + ratio * num_scales;
58 
59  for (int scale = 0; scale < num_scales; ++scale) {
60  // transformed width & height for given scale factors
61  const float scale_w = 0.5f * (ratio_w * scales[scale] - coordinates_offset);
62  const float scale_h = 0.5f * (ratio_h * scales[scale] - coordinates_offset);
63 
64  // (x1, y1, x2, y2) for transformed box
65  p_anchors_wm[scale] = center - scale_w;
66  p_anchors_hm[scale] = center - scale_h;
67  p_anchors_wp[scale] = center + scale_w;
68  p_anchors_hp[scale] = center + scale_h;
69 
70  if (shift_anchors) {
71  p_anchors_wm[scale] -= half_base_size;
72  p_anchors_hm[scale] -= half_base_size;
73  p_anchors_wp[scale] -= half_base_size;
74  p_anchors_hp[scale] -= half_base_size;
75  }
76  }
77  }
78 }

80 static
81 void enumerate_proposals_cpu(const float* bottom4d, const float* d_anchor4d, const float* anchors,
82  float* proposals, const int num_anchors, const int bottom_H,
83  const int bottom_W, const float img_H, const float img_W,
84  const float min_box_H, const float min_box_W, const int feat_stride,
85  const float box_coordinate_scale, const float box_size_scale,
86  float coordinates_offset, bool initial_clip, bool swap_xy) {
87  const int bottom_area = bottom_H * bottom_W;
88 
89  const float* p_anchors_wm = anchors + 0 * num_anchors;
90  const float* p_anchors_hm = anchors + 1 * num_anchors;
91  const float* p_anchors_wp = anchors + 2 * num_anchors;
92  const float* p_anchors_hp = anchors + 3 * num_anchors;
93 
94 #if _MSC_VER && !__INTEL_COMPILER
95  #pragma omp parallel for schedule(static)
96 #else
97  #pragma omp parallel for collapse(2) schedule(static)
98 #endif
99  for (int h = 0; h < bottom_H; ++h) {
100  for (int w = 0; w < bottom_W; ++w) {
101  const float x = (swap_xy ? h : w) * feat_stride;
102  const float y = (swap_xy ? w : h) * feat_stride;
103 
104  const float* p_box = d_anchor4d + h * bottom_W + w;
105  const float* p_score = bottom4d + h * bottom_W + w;
106 
107  float* p_proposal = proposals + (h * bottom_W + w) * num_anchors * 5;
108 
109  for (int anchor = 0; anchor < num_anchors; ++anchor) {
110  const float dx = p_box[(anchor * 4 + 0) * bottom_area] / box_coordinate_scale;
111  const float dy = p_box[(anchor * 4 + 1) * bottom_area] / box_coordinate_scale;
112 
113  const float d_log_w = p_box[(anchor * 4 + 2) * bottom_area] / box_size_scale;
114  const float d_log_h = p_box[(anchor * 4 + 3) * bottom_area] / box_size_scale;
115 
116  const float score = p_score[anchor * bottom_area];
117 
118  float x0 = x + p_anchors_wm[anchor];
119  float y0 = y + p_anchors_hm[anchor];
120  float x1 = x + p_anchors_wp[anchor];
121  float y1 = y + p_anchors_hp[anchor];
122 
123  if (initial_clip) {
124  // adjust new corner locations to be within the image region
125  x0 = std::max<float>(0.0f, std::min<float>(x0, img_W));
126  y0 = std::max<float>(0.0f, std::min<float>(y0, img_H));
127  x1 = std::max<float>(0.0f, std::min<float>(x1, img_W));
128  y1 = std::max<float>(0.0f, std::min<float>(y1, img_H));
129  }
130 
131  // width & height of box
132  const float ww = x1 - x0 + coordinates_offset;
133  const float hh = y1 - y0 + coordinates_offset;
134  // center location of box
135  const float ctr_x = x0 + 0.5f * ww;
136  const float ctr_y = y0 + 0.5f * hh;
137 
138  // new center location according to gradient (dx, dy)
139  const float pred_ctr_x = dx * ww + ctr_x;
140  const float pred_ctr_y = dy * hh + ctr_y;
141  // new width & height according to gradient d(log w), d(log h)
142  const float pred_w = std::exp(d_log_w) * ww;
143  const float pred_h = std::exp(d_log_h) * hh;
144 
145  // update upper-left corner location
146  x0 = pred_ctr_x - 0.5f * pred_w;
147  y0 = pred_ctr_y - 0.5f * pred_h;
148  // update lower-right corner location
149  x1 = pred_ctr_x + 0.5f * pred_w;
150  y1 = pred_ctr_y + 0.5f * pred_h;
151 
152  // adjust new corner locations to be within the image region,
153  x0 = std::max<float>(0.0f, std::min<float>(x0, img_W - coordinates_offset));
154  y0 = std::max<float>(0.0f, std::min<float>(y0, img_H - coordinates_offset));
155  x1 = std::max<float>(0.0f, std::min<float>(x1, img_W - coordinates_offset));
156  y1 = std::max<float>(0.0f, std::min<float>(y1, img_H - coordinates_offset));
157 
158  // recompute new width & height
159  const float box_w = x1 - x0 + coordinates_offset;
160  const float box_h = y1 - y0 + coordinates_offset;
161 
162  p_proposal[5*anchor + 0] = x0;
163  p_proposal[5*anchor + 1] = y0;
164  p_proposal[5*anchor + 2] = x1;
165  p_proposal[5*anchor + 3] = y1;
166  p_proposal[5*anchor + 4] = (min_box_W <= box_w) * (min_box_H <= box_h) * score;
167  }
168  }
169  }

182 static
183 void nms_cpu(const int num_boxes, int is_dead[],
184  const float* boxes, int index_out[], int* const num_out,
185  const int base_index, const float nms_thresh, const int max_num_out,
186  float coordinates_offset) {
187  const int num_proposals = num_boxes;
188  int count = 0;
189 
190  const float* x0 = boxes + 0 * num_proposals;
191  const float* y0 = boxes + 1 * num_proposals;
192  const float* x1 = boxes + 2 * num_proposals;
193  const float* y1 = boxes + 3 * num_proposals;
194 
195  memset(is_dead, 0, num_boxes * sizeof(int));
196 
197 #if defined(HAVE_AVX2)
198  __m256 vc_fone = _mm256_set1_ps(coordinates_offset);
199  __m256i vc_ione = _mm256_set1_epi32(1);
200  __m256 vc_zero = _mm256_set1_ps(0.0f);
201 
202  __m256 vc_nms_thresh = _mm256_set1_ps(nms_thresh);
203 #endif
204 
205  for (int box = 0; box < num_boxes; ++box) {
206  if (is_dead[box])
207  continue;
208 
209  index_out[count++] = base_index + box;
210  if (count == max_num_out)
211  break;
212 
213  int tail = box + 1;
214 
215 #if defined(HAVE_AVX2)
216  __m256 vx0i = _mm256_set1_ps(x0[box]);
217  __m256 vy0i = _mm256_set1_ps(y0[box]);
218  __m256 vx1i = _mm256_set1_ps(x1[box]);
219  __m256 vy1i = _mm256_set1_ps(y1[box]);
220 
221  __m256 vA_width = _mm256_sub_ps(vx1i, vx0i);
222  __m256 vA_height = _mm256_sub_ps(vy1i, vy0i);
223  __m256 vA_area = _mm256_mul_ps(_mm256_add_ps(vA_width, vc_fone), _mm256_add_ps(vA_height, vc_fone));
224 
225  for (; tail <= num_boxes - 8; tail += 8) {
226  __m256i *pdst = reinterpret_cast<__m256i*>(is_dead + tail);
227  __m256i vdst = _mm256_loadu_si256(pdst);
228 
229  __m256 vx0j = _mm256_loadu_ps(x0 + tail);
230  __m256 vy0j = _mm256_loadu_ps(y0 + tail);
231  __m256 vx1j = _mm256_loadu_ps(x1 + tail);
232  __m256 vy1j = _mm256_loadu_ps(y1 + tail);
233 
234  __m256 vx0 = _mm256_max_ps(vx0i, vx0j);
235  __m256 vy0 = _mm256_max_ps(vy0i, vy0j);
236  __m256 vx1 = _mm256_min_ps(vx1i, vx1j);
237  __m256 vy1 = _mm256_min_ps(vy1i, vy1j);
238 
239  __m256 vwidth = _mm256_add_ps(_mm256_sub_ps(vx1, vx0), vc_fone);
240  __m256 vheight = _mm256_add_ps(_mm256_sub_ps(vy1, vy0), vc_fone);
241  __m256 varea = _mm256_mul_ps(_mm256_max_ps(vc_zero, vwidth), _mm256_max_ps(vc_zero, vheight));
242 
243  __m256 vB_width = _mm256_sub_ps(vx1j, vx0j);
244  __m256 vB_height = _mm256_sub_ps(vy1j, vy0j);
245  __m256 vB_area = _mm256_mul_ps(_mm256_add_ps(vB_width, vc_fone), _mm256_add_ps(vB_height, vc_fone));
246 
247  __m256 vdivisor = _mm256_sub_ps(_mm256_add_ps(vA_area, vB_area), varea);
248  __m256 vintersection_area = _mm256_div_ps(varea, vdivisor);
249 
250  __m256 vcmp_0 = _mm256_cmp_ps(vx0i, vx1j, _CMP_LE_OS);
251  __m256 vcmp_1 = _mm256_cmp_ps(vy0i, vy1j, _CMP_LE_OS);
252  __m256 vcmp_2 = _mm256_cmp_ps(vx0j, vx1i, _CMP_LE_OS);
253  __m256 vcmp_3 = _mm256_cmp_ps(vy0j, vy1i, _CMP_LE_OS);
254  __m256 vcmp_4 = _mm256_cmp_ps(vc_nms_thresh, vintersection_area, _CMP_LT_OS);
255 
256  vcmp_0 = _mm256_and_ps(vcmp_0, vcmp_1);
257  vcmp_2 = _mm256_and_ps(vcmp_2, vcmp_3);
258  vcmp_4 = _mm256_and_ps(vcmp_4, vcmp_0);
259  vcmp_4 = _mm256_and_ps(vcmp_4, vcmp_2);
260 
261  _mm256_storeu_si256(pdst, _mm256_blendv_epi8(vdst, vc_ione, _mm256_castps_si256(vcmp_4)));
262  }
263 #endif
264 
265  for (; tail < num_boxes; ++tail) {
266  float res = 0.0f;
267 
268  const float x0i = x0[box];
269  const float y0i = y0[box];
270  const float x1i = x1[box];
271  const float y1i = y1[box];
272 
273  const float x0j = x0[tail];
274  const float y0j = y0[tail];
275  const float x1j = x1[tail];
276  const float y1j = y1[tail];
277 
278  if (x0i <= x1j && y0i <= y1j && x0j <= x1i && y0j <= y1i) {
279  // overlapped region (= box)
280  const float x0 = std::max<float>(x0i, x0j);
281  const float y0 = std::max<float>(y0i, y0j);
282  const float x1 = std::min<float>(x1i, x1j);
283  const float y1 = std::min<float>(y1i, y1j);
284 
285  // intersection area
286  const float width = std::max<float>(0.0f, x1 - x0 + coordinates_offset);
287  const float height = std::max<float>(0.0f, y1 - y0 + coordinates_offset);
288  const float area = width * height;
289 
290  // area of A, B
291  const float A_area = (x1i - x0i + coordinates_offset) * (y1i - y0i + coordinates_offset);
292  const float B_area = (x1j - x0j + coordinates_offset) * (y1j - y0j + coordinates_offset);
293 
294  // IoU
295  res = area / (A_area + B_area - area);
296  }
297 
298  if (nms_thresh < res)
299  is_dead[tail] = 1;
300  }
301  }
302 
303  *num_out = count;
304 }

307 void retrieve_rois_cpu(const int num_rois, const int item_index,
308  const int num_proposals,
309  const float* proposals, const int roi_indices[],
310  float* rois, int post_nms_topn_) {
311  const float *src_x0 = proposals + 0 * num_proposals;
312  const float *src_y0 = proposals + 1 * num_proposals;
313  const float *src_x1 = proposals + 2 * num_proposals;
314  const float *src_y1 = proposals + 3 * num_proposals;
315 
316  #pragma omp parallel for schedule(static)
317  for (int roi = 0; roi < num_rois; roi++) {
318  int index = roi_indices[roi];
319 
320  const float x0 = src_x0[index];
321  const float y0 = src_y0[index];
322  const float x1 = src_x1[index];
323  const float y1 = src_y1[index];
324 
325  rois[roi * 5 + 0] = item_index;
326  rois[roi * 5 + 1] = x0;
327  rois[roi * 5 + 2] = y0;
328  rois[roi * 5 + 3] = x1;
329  rois[roi * 5 + 4] = y1;
330  }
331 
332  if (num_rois < post_nms_topn_) {
333  for (int i = 5 * num_rois; i < 5 * post_nms_topn_; i++) {
334  rois[i] = 0.f;
335  }
336 
337  // marker at end of boxes list
338  rois[num_rois * 5 + 0] = -1;
339  }

344  explicit ProposalImpl(const CNNLayer *layer): ExtLayerBase(layer) {
345  try {
346  if (cnnLayer.insData.size() != 3 || cnnLayer.outData.size() != 1)
347  THROW_IE_EXCEPTION << "Incorrect number of input/output edges!";
348  feat_stride_ = static_cast<size_t>(cnnLayer.GetParamAsInt("feat_stride"));
349  base_size_ = static_cast<size_t>(cnnLayer.GetParamAsInt("base_size"));
350  min_size_ = static_cast<size_t>(cnnLayer.GetParamAsInt("min_size"));
351  pre_nms_topn_ = cnnLayer.GetParamAsInt("pre_nms_topn");
352  post_nms_topn_ = cnnLayer.GetParamAsInt("post_nms_topn");
353  nms_thresh_ = cnnLayer.GetParamAsFloat("nms_thresh");
354  box_coordinate_scale_ = cnnLayer.GetParamAsFloat("box_coordinate_scale", 1.0);
355  box_size_scale_ = cnnLayer.GetParamAsFloat("box_size_scale", 1.0);
356  scales = cnnLayer.GetParamAsFloats("scale", {});
357  ratios = cnnLayer.GetParamAsFloats("ratio", {});
358 
359  anchors_shape_0 = ratios.size() * scales.size();
360  anchors_.resize(anchors_shape_0 * 4);
361 
362  std::string framework_ = cnnLayer.GetParamAsString("framework", "");
363  if (framework_ == "tensorflow") {
364  coordinates_offset = 0.0f;
365  initial_clip = true;
366  shift_anchors = true;
367  round_ratios = false;
368  swap_xy = true;
369  } else {
370  coordinates_offset = 1.0f;
371  initial_clip = false;
372  shift_anchors = false;
373  round_ratios = true;
374  swap_xy = false;
375  }
376  generate_anchors(base_size_, &ratios[0], &scales[0], ratios.size(), scales.size(), &anchors_[0],
377  coordinates_offset, shift_anchors, round_ratios);
378 
379  roi_indices_.resize(post_nms_topn_);
380  addConfig({DataConfigurator(ConfLayout::PLN), DataConfigurator(ConfLayout::PLN), DataConfigurator(ConfLayout::PLN)},
381  {DataConfigurator(ConfLayout::PLN)});
382  } catch (InferenceEngine::details::InferenceEngineException &ex) {
383  errorMsg = ex.what();
384  }
385  }

387  StatusCode execute(std::vector<Blob::Ptr> &inputs, std::vector<Blob::Ptr> &outputs,
388  ResponseDesc *resp) noexcept override {
389  if (inputs.size() != 3 || outputs.empty()) {
390  if (resp) {
391  std::string errorMsg = "Incorrect number of input or output edges!";
392  errorMsg.copy(resp->msg, sizeof(resp->msg) - 1);
393  }
394  return GENERAL_ERROR;
395  }
396 
397  // Prepare memory
398  const float* p_bottom_item = inputs[0]->buffer();
399  const float* p_d_anchor_item = inputs[1]->buffer();
400  const float* p_img_info_cpu = inputs[2]->buffer();
401  float* p_roi_item = outputs[0]->buffer();
402 
403  size_t img_info_size = 1;
404  for (size_t i = 0; i < inputs[2]->getTensorDesc().getDims().size(); i++) {
405  img_info_size *= inputs[2]->getTensorDesc().getDims()[i];
406  }
407 
408  // No second output so ignoring this
409  // Dtype* p_score_item = (top.size() > 1) ? top[1]->mutable_cpu_data() : NULL;
410 
411  // bottom shape: (2 x num_anchors) x H x W
412  const int bottom_H = inputs[0]->getTensorDesc().getDims()[2];
413  const int bottom_W = inputs[0]->getTensorDesc().getDims()[3];
414 
415  // input image height & width
416  const float img_H = p_img_info_cpu[0];
417  const float img_W = p_img_info_cpu[1];
418 
419  // scale factor for height & width
420  const float scale_H = p_img_info_cpu[2];
421  const float scale_W = img_info_size > 3 ? p_img_info_cpu[3] : scale_H;
422 
423  // minimum box width & height
424  const float min_box_H = min_size_ * scale_H;
425  const float min_box_W = min_size_ * scale_W;
426 
427  // number of all proposals = num_anchors * H * W
428  const int num_proposals = anchors_shape_0 * bottom_H * bottom_W;
429 
430  // number of top-n proposals before NMS
431  const int pre_nms_topn = std::min<int>(num_proposals, pre_nms_topn_);
432 
433  // number of final RoIs
434  int num_rois = 0;
435 
436  // enumerate all proposals
437  // num_proposals = num_anchors * H * W
438  // (x1, y1, x2, y2, score) for each proposal
439  // NOTE: for bottom, only foreground scores are passed
440  struct ProposalBox {
441  float x0;
442  float y0;
443  float x1;
444  float y1;
445  float score;
446  };
447  std::vector<ProposalBox> proposals_(num_proposals);
448  std::vector<float> unpacked_boxes(4 * pre_nms_topn);
449  std::vector<int> is_dead(pre_nms_topn);
450 
451  // Execute
452  int nn = inputs[0]->getTensorDesc().getDims()[0];
453  for (int n = 0; n < nn; ++n) {
454  enumerate_proposals_cpu(p_bottom_item + num_proposals, p_d_anchor_item,
455  &anchors_[0], reinterpret_cast<float *>(&proposals_[0]),
456  anchors_shape_0, bottom_H, bottom_W, img_H, img_W,
457  min_box_H, min_box_W, feat_stride_,
458  box_coordinate_scale_, box_size_scale_,
459  coordinates_offset, initial_clip, swap_xy);
460  std::partial_sort(proposals_.begin(), proposals_.begin() + pre_nms_topn, proposals_.end(),
461  [](const ProposalBox& struct1, const ProposalBox& struct2) {
462  return (struct1.score > struct2.score);
463  });
464 
465  unpack_boxes(reinterpret_cast<float *>(&proposals_[0]), &unpacked_boxes[0], pre_nms_topn);
466  nms_cpu(pre_nms_topn, &is_dead[0], &unpacked_boxes[0], &roi_indices_[0], &num_rois, 0, nms_thresh_, post_nms_topn_, coordinates_offset);
467  retrieve_rois_cpu(num_rois, n, pre_nms_topn, &unpacked_boxes[0], &roi_indices_[0], p_roi_item, post_nms_topn_);
468  }
469 
470  return OK;
471  }
