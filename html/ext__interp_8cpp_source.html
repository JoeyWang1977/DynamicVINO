52  StatusCode execute(std::vector<Blob::Ptr>& inputs, std::vector<Blob::Ptr>& outputs,
53  ResponseDesc *resp) noexcept override {
54  int IN = static_cast<int>(inputs[0]->getTensorDesc().getDims()[0]);
55  int IC = static_cast<int>(inputs[0]->getTensorDesc().getDims()[1]);
56  int IH = static_cast<int>(inputs[0]->getTensorDesc().getDims()[2]);
57  int IW = static_cast<int>(inputs[0]->getTensorDesc().getDims()[3]);
58 
59  int OH = static_cast<int>(outputs[0]->getTensorDesc().getDims()[2]);
60  int OW = static_cast<int>(outputs[0]->getTensorDesc().getDims()[3]);
61 
62  int IH_pad = IH + pad_beg + pad_end;
63  int IW_pad = IW + pad_beg + pad_end;
64 
65  const auto *src_data = inputs[0]->buffer().as<const float *>();
66  auto *dst_data = outputs[0]->buffer().as<float *>();
67 
68  interpolate(IN, IC, src_data, -pad_beg, -pad_beg, IH_pad, IW_pad, IH, IW, dst_data, 0, 0, OH, OW, OH, OW);
69  return OK;
70  }

72 private:
73  int pad_beg;
74  int pad_end;
75  void interpolate(const int N, const int C,
76  const float *src, const int x1, const int y1,
77  const int IH_pad, const int IW_pad, const int IH, const int IW,
78  float *dst, const int x2, const int y2,
79  const int OH_pad, const int OW_pad, const int OH, const int OW) {
80  if (IH_pad == OH_pad && IW_pad == OW_pad) {
81  for (int i = 0; i < N * C * OH * OW; i++) {
82  dst[i] = src[i];
83  }
84  return;
85  }
86 
87  const float rh = (OH_pad > 1) ? static_cast<float>(IH_pad - 1) / (OH_pad - 1) : 0.0f;
88  const float rw = (OW_pad > 1) ? static_cast<float>(IW_pad - 1) / (OW_pad - 1) : 0.0f;
89 
90 #if defined(HAVE_AVX512F)
91  const int block_size = 16;
92 #else
93  const int block_size = 8;
94 #endif
95 
96  // Align channel number to block size to deal with channels padding in IE with multiple blobs
97  int CB = (C + block_size - 1) & (-block_size);
98 
99  int CH = (C + block_size - 1) / block_size;
100 
101 #if _MSC_VER && !__INTEL_COMPILER
102  #pragma omp parallel for schedule(static)
103 #else
104  #pragma omp parallel for collapse(3) schedule(static)
105 #endif
106  for (int n = 0; n < N; n++) {
107  for (int cb = 0; cb < CH; ++cb) {
108  for (int h = 0; h < OH_pad; ++h) {
109  const float *psrc = src + n * CB * IH * IW;
110 
111  float fh = rh * h;
112  int ih0 = static_cast<int>(fh);
113  int ih1 = (ih0 < IH_pad - 1) ? ih0 + 1 : ih0;
114 
115  float h_lambda0 = fh - ih0;
116  float h_lambda1 = 1.0f - h_lambda0;
117 
118  for (int w = 0; w < OW_pad; ++w) {
119  float fw = rw * w;
120  int iw0 = static_cast<int>(fw);
121  int iw1 = (iw0 < IW_pad - 1) ? iw0 + 1 : iw0;
122 
123  float w_lambda0 = fw - iw0;
124  float w_lambda1 = 1.0f - w_lambda0;
125 
126  const float *psrc00 =
127  psrc + cb * block_size * IW * IH + (y1 + ih0) * IW * block_size + (x1 + iw0) * block_size;
128  const float *psrc01 =
129  psrc + cb * block_size * IW * IH + (y1 + ih0) * IW * block_size + (x1 + iw1) * block_size;
130  const float *psrc10 =
131  psrc + cb * block_size * IW * IH + (y1 + ih1) * IW * block_size + (x1 + iw0) * block_size;
132  const float *psrc11 =
133  psrc + cb * block_size * IW * IH + (y1 + ih1) * IW * block_size + (x1 + iw1) * block_size;
134 
135  float *pdst = dst + n * CB * OH * OW + cb * block_size * OW * OH + (y2 + h) * OW * block_size +
136  (x2 + w) * block_size;
137 
138 #if defined(HAVE_AVX512F)
139  __m512 vwl0 = _mm512_set1_ps(w_lambda0);
140  __m512 vwl1 = _mm512_set1_ps(w_lambda1);
141  __m512 vhl0 = _mm512_set1_ps(h_lambda0);
142  __m512 vhl1 = _mm512_set1_ps(h_lambda1);
143  __m512 vsrc00 = _mm512_loadu_ps(psrc00);
144  __m512 vsrc01 = _mm512_loadu_ps(psrc01);
145  __m512 vsrc10 = _mm512_loadu_ps(psrc10);
146  __m512 vsrc11 = _mm512_loadu_ps(psrc11);
147 
148  __m512 vdst0 = _mm512_fmadd_ps(vwl1, vsrc00, _mm512_mul_ps(vwl0, vsrc01));
149  __m512 vdst1 = _mm512_fmadd_ps(vwl1, vsrc10, _mm512_mul_ps(vwl0, vsrc11));
150  __m512 vdst = _mm512_fmadd_ps(vhl1, vdst0, _mm512_mul_ps(vhl0, vdst1));
151 
152  _mm512_storeu_ps(pdst, vdst);
153 #elif defined(HAVE_AVX2)
154  __m256 vwl0 = _mm256_set1_ps(w_lambda0);
155  __m256 vwl1 = _mm256_set1_ps(w_lambda1);
156  __m256 vhl0 = _mm256_set1_ps(h_lambda0);
157  __m256 vhl1 = _mm256_set1_ps(h_lambda1);
158  __m256 vsrc00 = _mm256_loadu_ps(psrc00);
159  __m256 vsrc01 = _mm256_loadu_ps(psrc01);
160  __m256 vsrc10 = _mm256_loadu_ps(psrc10);
161  __m256 vsrc11 = _mm256_loadu_ps(psrc11);
162 
163  __m256 vdst0 = _mm256_fmadd_ps(vwl1, vsrc00, _mm256_mul_ps(vwl0, vsrc01));
164  __m256 vdst1 = _mm256_fmadd_ps(vwl1, vsrc10, _mm256_mul_ps(vwl0, vsrc11));
165  __m256 vdst = _mm256_fmadd_ps(vhl1, vdst0, _mm256_mul_ps(vhl0, vdst1));
166 
167  _mm256_storeu_ps(pdst, vdst);
168 #elif defined(HAVE_SSE)
169  __m128 vwl0 = _mm_set1_ps(w_lambda0);
170  __m128 vwl1 = _mm_set1_ps(w_lambda1);
171  __m128 vhl0 = _mm_set1_ps(h_lambda0);
172  __m128 vhl1 = _mm_set1_ps(h_lambda1);
173  for (int i = 0; i < block_size/4; i++) {
174  __m128 vsrc00 = _mm_loadu_ps(psrc00 + i*block_size/2);
175  __m128 vsrc01 = _mm_loadu_ps(psrc01 + i*block_size/2);
176  __m128 vsrc10 = _mm_loadu_ps(psrc10 + i*block_size/2);
177  __m128 vsrc11 = _mm_loadu_ps(psrc11 + i*block_size/2);
178 
179  __m128 vdst00 = _mm_mul_ps(vwl1, vsrc00);
180  __m128 vdst01 = _mm_mul_ps(vwl0, vsrc01);
181  __m128 vdst10 = _mm_mul_ps(vwl1, vsrc10);
182  __m128 vdst11 = _mm_mul_ps(vwl0, vsrc11);
183 
184  __m128 vdst0 = _mm_add_ps(vdst00, vdst01);
185  __m128 vdst1 = _mm_add_ps(vdst10, vdst11);
186 
187  __m128 vdst = _mm_add_ps(_mm_mul_ps(vhl1, vdst0), _mm_mul_ps(vhl0, vdst1));
188 
189  _mm_storeu_ps(pdst + i*block_size/2, vdst);
190  }
191 #else
192  for (int c = 0; c < block_size; ++c) {
193  pdst[c] = h_lambda1 * (w_lambda1 * psrc00[c] + w_lambda0 * psrc01[c]) +
194  h_lambda0 * (w_lambda1 * psrc10[c] + w_lambda0 * psrc11[c]);
195  }
196 #endif
197  }
198  }
199  }
200  }
201  }
