28 class PriorBoxImpl: public ExtLayerBase {
29 public:
30  explicit PriorBoxImpl(const CNNLayer *layer): ExtLayerBase(layer) {
31  try {
32  if (cnnLayer.insData.size() != 2 || cnnLayer.outData.empty())
33  THROW_IE_EXCEPTION << "Incorrect number of input/output edges!";
34  _offset = cnnLayer.GetParamAsFloat("offset");
35  _step = cnnLayer.GetParamAsFloat("step", 0);
36  _min_sizes = cnnLayer.GetParamAsFloats("min_size", {});
37  _max_sizes = cnnLayer.GetParamAsFloats("max_size", {});
38  _flip = static_cast<bool>(cnnLayer.GetParamAsInt("flip"));
39  _clip = static_cast<bool>(cnnLayer.GetParamAsInt("clip"));
40  _scale_all_sizes = static_cast<bool>(cnnLayer.GetParamAsInt("scale_all_sizes", 1));
41 
42  bool exist;
43 
44  _aspect_ratios.push_back(1.0f);
45 
46  const std::vector<float> aspect_ratios = cnnLayer.GetParamAsFloats("aspect_ratio", {});
47 
48  for (float aspect_ratio : aspect_ratios) {
49  exist = false;
50 
51  for (float _aspect_ratio : _aspect_ratios) {
52  if (fabs(aspect_ratio - _aspect_ratio) < 1e-6) {
53  exist = true;
54  break;
55  }
56  }
57 
58  if (exist) {
59  continue;
60  }
61 
62  _aspect_ratios.push_back(aspect_ratio);
63 
64  if (_flip) {
65  _aspect_ratios.push_back(1.0f / aspect_ratio);
66  }
67  }
68 
69  if (_scale_all_sizes) {
70  _num_priors = static_cast<int>(_aspect_ratios.size() * _min_sizes.size());
71  } else {
72  _num_priors = static_cast<int>(_aspect_ratios.size() + _min_sizes.size() - 1);
73  }
74 
75  for (auto it = _max_sizes.begin(); it != _max_sizes.end(); it++) {
76  _num_priors += 1;
77  }
78 
79  const std::vector<float> variance = cnnLayer.GetParamAsFloats("variance", {});
80 
81  if (variance.size() == 1 || variance.size() == 4) {
82  for (float i : variance) {
83  if (i < 0) {
84  THROW_IE_EXCEPTION << "Variance must be > 0.";
85  }
86 
87  _variance.push_back(i);
88  }
89  } else if (variance.empty()) {
90  _variance.push_back(0.1f);
91  } else {
92  THROW_IE_EXCEPTION << "Wrong number of variance values. Not less than 1 and more than 4 variance values.";
93  }
94 
95  addConfig({{ConfLayout::ANY, true}, {ConfLayout::ANY, true}}, {{ConfLayout::PLN, true}});
96  } catch (InferenceEngine::details::InferenceEngineException &ex) {
97  errorMsg = ex.what();
98  }
99  }

101  StatusCode execute(std::vector<Blob::Ptr>& inputs, std::vector<Blob::Ptr>& outputs,
102  ResponseDesc *resp) noexcept override {
103  if (inputs.size() != 2 || outputs.empty()) {
104  if (resp) {
105  std::string errorMsg = "Incorrect number of input or output edges!";
106  errorMsg.copy(resp->msg, sizeof(resp->msg) - 1);
107  }
108  return GENERAL_ERROR;
109  }
110  auto& dataMemPtr = inputs[0];
111  auto& imageMemPtr = inputs[1];
112  auto& dstMemPtr = outputs[0];
113  SizeVector _data_dims = dataMemPtr->getTensorDesc().getDims();
114  SizeVector _image_dims = imageMemPtr->getTensorDesc().getDims();
115  const int W = _data_dims[3];
116  const int H = _data_dims[2];
117  const int IW = _image_dims[3];
118  const int IH = _image_dims[2];
119 
120  const int OH = dstMemPtr->getTensorDesc().getDims()[2];
121  const int OW = (dstMemPtr->getTensorDesc().getDims().size() == 3) ? 1 : dstMemPtr->getTensorDesc().getDims()[3];
122 
123  float step_x = 0.0f;
124  float step_y = 0.0f;
125 
126  if (_step == 0) {
127  step_x = static_cast<float>(IW) / W;
128  step_y = static_cast<float>(IH) / H;
129  } else {
130  step_x = _step;
131  step_y = _step;
132  }
133 
134  float* dst_data = dstMemPtr->buffer();
135 
136  int dim = H * W * _num_priors * 4;
137  int idx = 0;
138  float center_x = 0.0f;
139  float center_y = 0.0f;
140 
141  float box_width;
142  float box_height;
143 
144  for (int h = 0; h < H; ++h) {
145  for (int w = 0; w < W; ++w) {
146  for (size_t msIdx = 0; msIdx < _min_sizes.size(); msIdx++) {
147  if (_step == 0) {
148  center_x = (w + 0.5f) * step_x;
149  center_y = (h + 0.5f) * step_y;
150  } else {
151  center_x = (_offset + w) * _step;
152  center_y = (_offset + h) * _step;
153  }
154 
155  box_width = _min_sizes[msIdx];
156  box_height = _min_sizes[msIdx];
157 
158  dst_data[idx++] = (center_x - box_width / 2.0f) / IW;
159  dst_data[idx++] = (center_y - box_height / 2.0f) / IH;
160  dst_data[idx++] = (center_x + box_width / 2.0f) / IW;
161  dst_data[idx++] = (center_y + box_height / 2.0f) / IH;
162 
163  if (_max_sizes.size() > msIdx) {
164  box_width = box_height = sqrt(_min_sizes[msIdx] * _max_sizes[msIdx]);
165 
166  dst_data[idx++] = (center_x - box_width / 2.0f) / IW;
167  dst_data[idx++] = (center_y - box_height / 2.0f) / IH;
168  dst_data[idx++] = (center_x + box_width / 2.0f) / IW;
169  dst_data[idx++] = (center_y + box_height / 2.0f) / IH;
170  }
171 
172  if (_scale_all_sizes || (!_scale_all_sizes && (msIdx == _min_sizes.size() - 1))) {
173  size_t sIdx = _scale_all_sizes ? msIdx : 0;
174  for (float ar : _aspect_ratios) {
175  if (fabs(ar - 1.0f) < 1e-6) {
176  continue;
177  }
178 
179  box_width = _min_sizes[sIdx] * sqrt(ar);
180  box_height = _min_sizes[sIdx] / sqrt(ar);
181 
182  dst_data[idx++] = (center_x - box_width / 2.0f) / IW;
183  dst_data[idx++] = (center_y - box_height / 2.0f) / IH;
184  dst_data[idx++] = (center_x + box_width / 2.0f) / IW;
185  dst_data[idx++] = (center_y + box_height / 2.0f) / IH;
186  }
187  }
188  }
189  }
190  }
191 
192  if (_clip) {
193  for (int d = 0; d < dim; ++d) {
194  dst_data[d] = (std::min)((std::max)(dst_data[d], 0.0f), 1.0f);
195  }
196  }
197 
198  int channel_size = OH * OW;
199 
200  dst_data += channel_size;
201 
202  int count = 0;
203  if (_variance.size() == 1) {
204  for (int i = 0; i < channel_size; i++) {
205  dst_data[i] = _variance[0];
206  }
207  } else {
208  for (int h = 0; h < H; ++h) {
209  for (int w = 0; w < W; ++w) {
210  for (int i = 0; i < _num_priors; ++i) {
211  for (int j = 0; j < 4; ++j) {
212  dst_data[count] = _variance[j];
213  ++count;
214  }
215  }
216  }
217  }
218  }
219  return OK;
220  }
