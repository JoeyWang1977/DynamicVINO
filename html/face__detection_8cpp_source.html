47 bool openvino_service::FaceDetection::fetchResults() {
48  bool can_fetch = openvino_service::BaseInference::fetchResults();
49  if (!can_fetch) return false;
50  bool found_result = false;
51  results_.clear();
52  InferenceEngine::InferRequest::Ptr request = getEngine()->getRequest();
53  std::string output = valid_network_->getOutputName();
54  const float *detections = request->GetBlob(output)->buffer().as<float *>();
55  for (int i = 0; i < max_proposal_count_; i++) {
56  float image_id = detections[i * object_size_ + 0];
57  Result r;
58  auto label_num = static_cast<int>(detections[i * object_size_ + 1]);
59  std::vector<std::string> &labels = valid_network_->getLabels();
60  r.label = label_num < labels.size() ? labels[label_num] :
61  std::string("label #") + std::to_string(label_num);
62  r.confidence = detections[i * object_size_ + 2];
63  if (r.confidence <= show_output_thresh_) {
64  continue;
65  }
66  found_result = true;
67  r.location.x = static_cast<int>(detections[i * object_size_ + 3] *
68  width_);
69  r.location.y = static_cast<int>(detections[i * object_size_ + 4] *
70  height_);
71  r.location.width = static_cast<int>(
72  detections[i * object_size_ + 5] * width_ - r.location.x);
73  r.location.height = static_cast<int>(
74  detections[i * object_size_ + 6] * height_ - r.location.y);
75 
76  if (image_id < 0) {
77  break;
78  }
79  results_.emplace_back(r);
80  }
81  if (!found_result) results_.clear();
82  return true;
83 };
