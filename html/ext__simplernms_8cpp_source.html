71 static void MakeAnchors(const std::vector<float>& ws, const std::vector<float>& hs,
72  float x_center, float y_center, std::vector<simpler_nms_anchor>& anchors) {
73  unsigned int len = ws.size();
74  anchors.clear();
75  anchors.resize(len);
76 
77  for (unsigned int i = 0 ; i < len ; i++) {
78  // transpose to create the anchor
79  anchors[i].start_x = x_center - 0.5f * (ws[i] - 1.0f);
80  anchors[i].start_y = y_center - 0.5f * (hs[i] - 1.0f);
81  anchors[i].end_x = x_center + 0.5f * (ws[i] - 1.0f);
82  anchors[i].end_y = y_center + 0.5f * (hs[i] - 1.0f);
83  }
84 }

87 static void CalcAnchors(const simpler_nms_anchor& base_anchor, const std::vector<float>& scales,
88  std::vector<simpler_nms_anchor>& anchors) {
89  float width = 0.0f, height = 0.0f, x_center = 0.0f, y_center = 0.0f;
90 
91  CalcBasicParams(base_anchor, width, height, x_center, y_center);
92 
93  unsigned int num_scales = scales.size();
94  std::vector<float> ws(num_scales), hs(num_scales);
95 
96  for (unsigned int i = 0 ; i < num_scales ; i++) {
97  ws[i] = width * scales[i];
98  hs[i] = height * scales[i];
99  }
100 
101  MakeAnchors(ws, hs, x_center, y_center, anchors);
102 }

105 static void CalcRatioAnchors(const simpler_nms_anchor& base_anchor, const std::vector<float>& ratios,
106  std::vector<simpler_nms_anchor>& ratio_anchors) {
107  float width = 0.0f, height = 0.0f, x_center = 0.0f, y_center = 0.0f;
108 
109  CalcBasicParams(base_anchor, width, height, x_center, y_center);
110 
111  float size = width * height;
112 
113  unsigned int num_ratios = ratios.size();
114 
115  std::vector<float> ws(num_ratios), hs(num_ratios);
116 
117  for (unsigned int i = 0 ; i < num_ratios ; i++) {
118  float new_size = size / ratios[i];
119  ws[i] = round(sqrt(new_size));
120  hs[i] = round(ws[i] * ratios[i]);
121  }
122 
123  MakeAnchors(ws, hs, x_center, y_center, ratio_anchors);
124 }

126 void GenerateAnchors(unsigned int base_size, const std::vector<float>& ratios,
127  const std::vector<float> scales, simpler_nms_anchor *anchors) {
128  float end = static_cast<float>(base_size - 1); // because we start at zero
129 
130  simpler_nms_anchor base_anchor = {0.0f, 0.0f, end, end};
131 
132  std::vector<simpler_nms_anchor> ratio_anchors;
133  CalcRatioAnchors(base_anchor, ratios, ratio_anchors);
134 
135  for (size_t i = 0, index = 0; i < ratio_anchors.size() ; i++) {
136  std::vector<simpler_nms_anchor> temp_anchors;
137  CalcAnchors(ratio_anchors[i], scales, temp_anchors);
138 
139  for (size_t j = 0 ; j < temp_anchors.size() ; j++) {
140  anchors[index++] = temp_anchors[j];
141  }
142  }
143 }

145 std::vector<simpler_nms_roi_t> simpler_nms_perform_nms(
146  const std::vector<simpler_nms_proposal_t>& proposals,
147  float iou_threshold,
148  size_t top_n) {
149  std::vector<simpler_nms_roi_t> res;
150  res.reserve(top_n);
151  for (const auto & prop : proposals) {
152  const auto bbox = prop.roi;
153  const float area = bbox.area();
154 
155  // For any realistic WL, this condition is true for all top_n values anyway
156  if (prop.confidence > 0) {
157  bool overlaps = std::any_of(res.begin(), res.end(), [&](const simpler_nms_roi_t& res_bbox) {
158  float interArea = bbox.intersect(res_bbox).area();
159  float unionArea = res_bbox.area() + area - interArea;
160  return interArea > iou_threshold * unionArea;
161  });
162 
163  if (!overlaps) {
164  res.push_back(bbox);
165  if (res.size() == top_n) break;
166  }
167  }
168  }
169 
170  return res;
171 }

173 inline void sort_and_keep_at_most_top_n(
174  std::vector<simpler_nms_proposal_t>& proposals,
175  size_t top_n) {
176  const auto cmp_fn = [](const simpler_nms_proposal_t& a,
177  const simpler_nms_proposal_t& b) {
178  return a.confidence > b.confidence || (a.confidence == b.confidence && a.ord > b.ord);
179  };
180 
181  if (proposals.size() > top_n) {
182  std::partial_sort(proposals.begin(), proposals.begin() + top_n, proposals.end(), cmp_fn);
183  proposals.resize(top_n);
184  } else {
185  std::sort(proposals.begin(), proposals.end(), cmp_fn);
186  }
187 }

189 inline simpler_nms_roi_t simpler_nms_gen_bbox(
190  const simpler_nms_anchor& box,
191  const simpler_nms_delta_t& delta,
192  int anchor_shift_x,
193  int anchor_shift_y) {
194  auto anchor_w = box.end_x - box.start_x + 1;
195  auto anchor_h = box.end_y - box.start_y + 1;
196  auto center_x = box.start_x + anchor_w * .5f;
197  auto center_y = box.start_y + anchor_h *.5f;
198 
199  float pred_center_x = delta.shift_x * anchor_w + center_x + anchor_shift_x;
200  float pred_center_y = delta.shift_y * anchor_h + center_y + anchor_shift_y;
201  float half_pred_w = exp(delta.log_w) * anchor_w * .5f;
202  float half_pred_h = exp(delta.log_h) * anchor_h * .5f;
203 
204  return { pred_center_x - half_pred_w,
205  pred_center_y - half_pred_h,
206  pred_center_x + half_pred_w,
207  pred_center_y + half_pred_h };
208 }

216

218  feat_stride_ = cnnLayer.GetParamAsInt("feat_stride");
219  pre_nms_topn_ = cnnLayer.GetParamAsInt("pre_nms_topn");
220  post_nms_topn_ = cnnLayer.GetParamAsInt("post_nms_topn");
221  iou_threshold_ = cnnLayer.GetParamAsFloat("iou_threshold");
222  scales = cnnLayer.GetParamAsFloats("scale", {});

224  unsigned int default_size = 16;
225 
226  ratios = {0.5f, 1.0f, 2.0f};
227 
228  anchors_.resize(ratios.size() * scales.size());
229  simpler_nms_anchor *anchors = &anchors_[0];
230 
231  GenerateAnchors(default_size, ratios, scales, anchors);
232 
233  // Fill config information
234  if (cnnLayer.outData[0]->getTensorDesc().getDims().size() != 2 ||
235  cnnLayer.insData[0].lock()->getTensorDesc().getDims().size() != 4)
236  THROW_IE_EXCEPTION << "Unsupported dimensions!";
237 
238  addConfig({DataConfigurator(ConfLayout::PLN), DataConfigurator(ConfLayout::PLN), DataConfigurator(ConfLayout::PLN)},
239  {DataConfigurator(ConfLayout::PLN)});
240  } catch (InferenceEngine::details::InferenceEngineException &ex) {

245  StatusCode execute(std::vector<Blob::Ptr> &inputs, std::vector<Blob::Ptr> &outputs,
246  ResponseDesc *resp) noexcept override {
247  int cls_idx = 0;
248  int delta_idx = 1;
249 
250  Blob::Ptr src_cls = inputs[cls_idx];
251  Blob::Ptr src_delta = inputs[delta_idx];
252 
253  if (src_cls->getTensorDesc().getDims()[1] > src_delta->getTensorDesc().getDims()[1]) {
254  cls_idx = 1;
255  delta_idx = 0;
256 
257  src_cls = inputs[cls_idx];
258  src_delta = inputs[delta_idx];
259  }
260 
261  int anchors_num = 3 * 3;
262  const auto * anchors = (const simpler_nms_anchor*)&anchors_[0];
263 
264  int H = src_cls->getTensorDesc().getDims()[2];
265  int W = src_cls->getTensorDesc().getDims()[3];
266 
267  int SZ = H * W;
268 
269  float *dst = outputs[0]->buffer().as<float*>();
270 
271  const float* cls_scores = src_cls->buffer().as<const float*>();
272  const float* delta_pred = src_delta->buffer().as<const float*>();
273  const float* im_info = inputs[2]->buffer().as<const float*>();
274 
275  int IW = im_info[1];
276  int IH = im_info[0];
277  int IS = im_info[2];
278 
279  int scaled_min_bbox_size = min_box_size_ * IS;
280 
281  std::vector<simpler_nms_proposal_t> sorted_proposals_confidence;
282 
283  for (auto y = 0; y < H; ++y) {
284  int anchor_shift_y = y * feat_stride_;
285 
286  for (auto x = 0; x < W; ++x) {
287  int anchor_shift_x = x * feat_stride_;
288  int location_index = y * W + x;
289 
290  // we assume proposals are grouped by window location
291  for (int anchor_index = 0; anchor_index < anchors_num ; anchor_index++) {
292  float dx0 = delta_pred[location_index + SZ * (anchor_index * 4 + 0)];
293  float dy0 = delta_pred[location_index + SZ * (anchor_index * 4 + 1)];
294  float dx1 = delta_pred[location_index + SZ * (anchor_index * 4 + 2)];
295  float dy1 = delta_pred[location_index + SZ * (anchor_index * 4 + 3)];
296 
297  simpler_nms_delta_t bbox_delta { dx0, dy0, dx1, dy1 };
298 
299  float proposal_confidence =
300  cls_scores[location_index + SZ * (anchor_index + anchors_num * 1)];
301 
302  simpler_nms_roi_t tmp_roi = simpler_nms_gen_bbox(anchors[anchor_index], bbox_delta, anchor_shift_x, anchor_shift_y);
303  simpler_nms_roi_t roi = tmp_roi.clamp({ 0, 0, static_cast<float>(IW - 1), static_cast<float>(IH - 1)});
304 
305  int bbox_w = roi.x1 - roi.x0 + 1;
306  int bbox_h = roi.y1 - roi.y0 + 1;
307 
308  if (bbox_w >= scaled_min_bbox_size && bbox_h >= scaled_min_bbox_size) {
309  simpler_nms_proposal_t proposal { roi, proposal_confidence, sorted_proposals_confidence.size() };
310  sorted_proposals_confidence.push_back(proposal);
311  }
312  }
313  }
314  }
315 
316  sort_and_keep_at_most_top_n(sorted_proposals_confidence, pre_nms_topn_);
317  auto res = simpler_nms_perform_nms(sorted_proposals_confidence, iou_threshold_, post_nms_topn_);
318 
319  size_t res_num_rois = res.size();
320 
321  for (size_t i = 0; i < res_num_rois; ++i) {
322  dst[5 * i + 0] = 0; // roi_batch_ind, always zero on test time
323  dst[5 * i + 1] = res[i].x0;
324  dst[5 * i + 2] = res[i].y0;
325  dst[5 * i + 3] = res[i].x1;
326  dst[5 * i + 4] = res[i].y1;
327  }
328  return OK;
329  }
