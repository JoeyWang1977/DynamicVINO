50 #if defined(HAVE_SSE) || defined(HAVE_AVX2)
51  float hsum_sse(__m128 v) {
52  __m128 shuf = _mm_movehdup_ps(v);
53  __m128 sum = _mm_add_ps(v, shuf);
54  shuf = _mm_movehl_ps(shuf, sum);
55  sum = _mm_add_ss(sum, shuf);

73  ResponseDesc *resp) noexcept override {

99  if (across_spatial) {
100  float norm = eps;
101  int i = 0;
102 #if defined(HAVE_AVX2)
103  {
104  __m256 vsum = _mm256_setzero_ps();
105  for (; i <= C*H*W-8; i += 8) {
106  __m256 vsrc = _mm256_loadu_ps(psrc + i);
107  vsum = _mm256_fmadd_ps(vsrc, vsrc, vsum);
108  }
109  norm += hsum_avx2(vsum);
110  }
111 #elif defined(HAVE_SSE)
112  {
113  __m128 vsum = _mm_setzero_ps();
114  for (; i <= C*H*W-4; i += 4) {
115  __m128 vsrc = _mm_loadu_ps(psrc + i);
116  vsum = _mm_add_ps(_mm_mul_ps(vsrc, vsrc), vsum);
117  }
118  norm += hsum_sse(vsum);
119  }
120 #endif

122  norm += psrc[i]*psrc[i];
123  }

136  }
137 #elif defined(HAVE_SSE)
138  __m128 vnorm_sse = _mm_set1_ps(norm);

146 #endif

150  }
151  }
152  } else {
153  int wh = 0;
154 #if defined(HAVE_AVX2)
155  for (; wh <= W*H - 8; wh += 8) {
156  __m256 vnorm = _mm256_set1_ps(eps);
157  for (int c = 0; c < C; c++) {
158  const float* psrc_c = psrc + c*W*H;

160  vnorm = _mm256_fmadd_ps(vsrc, vsrc, vnorm);
161  }

163 
164  for (int c = 0; c < C; c++) {

166  float* pdst_c = pdst + c*W*H;
167

169 
170  __m256 vsrc = _mm256_loadu_ps(psrc_c + wh);
171  __m256 vdst = _mm256_mul_ps(vsrc, vnorm);
172  vdst = _mm256_mul_ps(vdst, vscl);
173 
174  _mm256_storeu_ps(pdst_c + wh, vdst);
175  }
176  }
177 #elif defined(HAVE_SSE)
178  for (; wh <= W*H - 4; wh += 4) {
179  __m128 vnorm = _mm_set1_ps(eps);
180  for (int c = 0; c < C; c++) {
181  const float* psrc_c = psrc + c*W*H;
182  __m128 vsrc = _mm_loadu_ps(psrc_c + wh);
183

191  float* pdst_c = pdst + c*W*H;

199  _mm_storeu_ps(pdst_c + wh, vdst);

201  }
202 #endif
203  for (; wh < W*H; wh++) {
204  float norm = eps;
205  for (int c = 0; c < C; c++) {
206  const float* psrc_c = psrc + c*W*H;
207  norm += psrc_c[wh]*psrc_c[wh];
208  }

214  float* pdst_c = pdst + c*W*H;
