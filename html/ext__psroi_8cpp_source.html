30  explicit PSROIPoolingImpl(const CNNLayer* layer): ExtLayerBase(layer) {
31  try {
32  if (cnnLayer.insData.size() != 2 || cnnLayer.outData.size() != 1)
33  THROW_IE_EXCEPTION << "Incorrect number of input/output edges!";
34  // LayerSetUp
35  output_dim_ = static_cast<size_t>(cnnLayer.GetParamAsInt("output_dim"));
36  group_size_ = static_cast<size_t>(cnnLayer.GetParamAsInt("group_size"));
37  spatial_scale_ = cnnLayer.GetParamAsFloat("spatial_scale");
38  pooled_height_ = group_size_;
39  pooled_width_ = group_size_;
40 
41  SizeVector inDims = cnnLayer.insData[0].lock()->getTensorDesc().getDims();
42  channels = static_cast<int>(inDims[1]);
43  height = static_cast<int>(inDims[2]);
44  width = static_cast<int>(inDims[3]);
45 
46  SizeVector outDims = cnnLayer.outData[0]->getTensorDesc().getDims();
47  nn = static_cast<int>(outDims[0]);
48  nc = static_cast<int>(outDims[1]);
49  nh = static_cast<int>(outDims[2]);
50  nw = static_cast<int>(outDims[3]);
51 
52  addConfig({DataConfigurator(ConfLayout::PLN), DataConfigurator(ConfLayout::PLN)}, {DataConfigurator(ConfLayout::PLN)});

55  }

58  StatusCode execute(std::vector<Blob::Ptr>& inputs, std::vector<Blob::Ptr>& outputs,
59  ResponseDesc *resp) noexcept override {
60  float* dst_data = outputs[0]->buffer();
61  const float *bottom_data_beginning = inputs[0]->buffer();
62  const float *bottom_rois_beginning = inputs[1]->buffer();
63 
64  int real_rois = 0;
65  for (; real_rois < nn; real_rois++) {
66  const float *bottom_rois = bottom_rois_beginning + real_rois * 5;
67  int roi_batch_ind = static_cast<int>(bottom_rois[0]);
68  if (roi_batch_ind == -1) {
69  break;
70  }
71  }
72 
73  #pragma omp parallel for schedule(static)
74  for (int n = 0; n < real_rois; n++) {
75  const float* bottom_rois = bottom_rois_beginning + n * 5;
76  int roi_batch_ind = static_cast<int>(bottom_rois[0]);
77  float roi_start_w = static_cast<float>(round(bottom_rois[1])) * spatial_scale_;
78  float roi_start_h = static_cast<float>(round(bottom_rois[2])) * spatial_scale_;
79  float roi_end_w = static_cast<float>(round(bottom_rois[3]) + 1.0f) * spatial_scale_;
80  float roi_end_h = static_cast<float>(round(bottom_rois[4]) + 1.0f) * spatial_scale_;
81 
82  // Force too small ROIs to be 1x1
83  float roi_width = std::max<float>(roi_end_w - roi_start_w, 0.1f); // avoid 0
84  float roi_height = std::max<float>(roi_end_h - roi_start_h, 0.1f);
85 
86  float bin_size_h = roi_height / static_cast<float>(pooled_height_);
87  float bin_size_w = roi_width / static_cast<float>(pooled_width_);
88 
89  for (int c = 0; c < nc; c++) {
90  for (int h = 0; h < nh; h++) {
91  int hstart = floor(static_cast<float>(h + 0) * bin_size_h + roi_start_h);
92  int hend = ceil(static_cast<float>(h + 1) * bin_size_h + roi_start_h);
93 
94  hstart = std::min<int>(std::max<int>(hstart, 0), height);
95  hend = std::min<int>(std::max<int>(hend, 0), height);
96 
97  for (int w = 0; w < nw; w++) {
98  int index = n * nc * nh * nw + c * nh * nw + h * nw + w;
99  dst_data[index] = 0.0f;
100 
101  int wstart = floor(static_cast<float>(w + 0) * bin_size_w + roi_start_w);
102  int wend = ceil(static_cast<float>(w + 1) * bin_size_w + roi_start_w);
103 
104  wstart = std::min<int>(std::max<int>(wstart, 0), width);
105  wend = std::min<int>(std::max<int>(wend, 0), width);
106 
107  float bin_area = (hend - hstart) * (wend - wstart);
108  if (bin_area) {
109  int gc = (c * group_size_ + h) * group_size_ + w;
110  const float *bottom_data =
111  bottom_data_beginning + ((roi_batch_ind * channels + gc) * height * width);
112 
113  float out_sum = 0.0f;
114  for (int hh = hstart; hh < hend; ++hh)
115  for (int ww = wstart; ww < wend; ++ww)
116  out_sum += bottom_data[hh * width + ww];
117 
118  dst_data[index] = out_sum / bin_area;
119  }
120  }
121  }
122  }
123  }
124 
125  #pragma omp parallel for schedule(static)
126  for (int n = real_rois; n < nn; n++) {
127  for (int c = 0; c < nc; c++) {
128  for (int h = 0; h < nh; h++) {
129  for (int w = 0; w < nw; w++) {
130  int index = n * nc * nh * nw + c * nh * nw + h * nw + w;
131  dst_data[index] = 0.0f;
132  }
133  }
134  }
135  }
136 
137  return OK;
138  }
