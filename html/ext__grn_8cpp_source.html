54 
55 #if _MSC_VER && !__INTEL_COMPILER
56  #pragma omp parallel for schedule(static)
57 #else
58  #pragma omp parallel for collapse(3) schedule(static)
59 #endif
60  for (int b = 0; b < N; b++) {
61  for (int h = 0; h < H; h++) {
62  for (int w = 0; w < W; w++) {
63  double variance = 0;
64  for (int c = 0; c < C; c++) {
65  variance += std::pow(src_data[b*C*H*W + c*H*W + h*W + w], 2);
66  }
67  variance = std::pow(variance + bias, 0.5f);
68  for (int c = 0; c < C; c++) {
69  dst_data[b*C*H*W + c*H*W + h*W + w] = src_data[b*C*H*W + c*H*W + h*W + w] / variance;
70  }
