31 // Revamped and reorganized by Craig Silverstein
32 //
33 // This file contains the implementation of all our command line flags
34 // stuff. Here's how everything fits together
35 //
36 // * FlagRegistry owns CommandLineFlags owns FlagValue.
37 // * FlagSaver holds a FlagRegistry (saves it at construct time,
38 // restores it at destroy time).
39 // * CommandLineFlagParser lives outside that hierarchy, but works on
40 // CommandLineFlags (modifying the FlagValues).
41 // * Free functions like SetCommandLineOption() work via one of the
42 // above (such as CommandLineFlagParser).
43 //
44 // In more detail:
45 //
46 // -- The main classes that hold flag data:
47 //
48 // FlagValue holds the current value of a flag. It's
49 // pseudo-templatized: every operation on a FlagValue is typed. It
50 // also deals with storage-lifetime issues (so flag values don't go
51 // away in a destructor), which is why we need a whole class to hold a
52 // variable's value.
53 //
54 // CommandLineFlag is all the information about a single command-line
55 // flag. It has a FlagValue for the flag's current value, but also
56 // the flag's name, type, etc.
57 //
58 // FlagRegistry is a collection of CommandLineFlags. There's the
59 // global registry, which is where flags defined via DEFINE_foo()
60 // live. But it's possible to define your own flag, manually, in a
61 // different registry you create. (In practice, multiple registries
62 // are used only by FlagSaver).
63 //
64 // A given FlagValue is owned by exactly one CommandLineFlag. A given
65 // CommandLineFlag is owned by exactly one FlagRegistry. FlagRegistry
66 // has a lock; any operation that writes to a FlagValue or
67 // CommandLineFlag owned by that registry must acquire the
68 // FlagRegistry lock before doing so.
69 //
70 // --- Some other classes and free functions:
71 //
72 // CommandLineFlagInfo is a client-exposed version of CommandLineFlag.
73 // Once it's instantiated, it has no dependencies or relationships
74 // with any other part of this file.
75 //
76 // FlagRegisterer is the helper class used by the DEFINE_* macros to
77 // allow work to be done at global initialization time.
78 //
79 // CommandLineFlagParser is the class that reads from the commandline
80 // and instantiates flag values based on that. It needs to poke into
81 // the innards of the FlagValue->CommandLineFlag->FlagRegistry class
82 // hierarchy to do that. It's careful to acquire the FlagRegistry
83 // lock before doing any writing or other non-const actions.
84 //
85 // GetCommandLineOption is just a hook into registry routines to
86 // retrieve a flag based on its name. SetCommandLineOption, on the
87 // other hand, hooks into CommandLineFlagParser. Other API functions
88 // are, similarly, mostly hooks into the functionality described above.

293 bool FlagValue::ParseFrom(const char* value) {
294  if (type_ == FV_BOOL) {
295  const char* kTrue[] = { "1", "t", "true", "y", "yes" };
296  const char* kFalse[] = { "0", "f", "false", "n", "no" };
297  COMPILE_ASSERT(sizeof(kTrue) == sizeof(kFalse), true_false_equal);
298  for (size_t i = 0; i < sizeof(kTrue)/sizeof(*kTrue); ++i) {
299  if (strcasecmp(value, kTrue[i]) == 0) {
300  SET_VALUE_AS(bool, true);
301  return true;
302  } else if (strcasecmp(value, kFalse[i]) == 0) {
303  SET_VALUE_AS(bool, false);
304  return true;
305  }
306  }
307  return false; // didn't match a legal input
308 
309  } else if (type_ == FV_STRING) {
310  SET_VALUE_AS(string, value);
311  return true;
312  }
313 
314  // OK, it's likely to be numeric, and we'll be using a strtoXXX method.
315  if (value[0] == '\0') // empty-string is only allowed for string type.
316  return false;
317  char* end;
318  // Leading 0x puts us in base 16. But leading 0 does not put us in base 8!
319  // It caused too many bugs when we had that behavior.
320  int base = 10; // by default
321  if (value[0] == '0' && (value[1] == 'x' || value[1] == 'X'))
322  base = 16;
323  errno = 0;
324 
325  switch (type_) {
326  case FV_INT32: {
327  const int64 r = strto64(value, &end, base);
328  if (errno || end != value + strlen(value)) return false; // bad parse
329  if (static_cast<int32>(r) != r) // worked, but number out of range
330  return false;
331  SET_VALUE_AS(int32, static_cast<int32>(r));
332  return true;
333  }
334  case FV_UINT32: {
335  while (*value == ' ') value++;
336  if (*value == '-') return false; // negative number
337  const uint64 r = strtou64(value, &end, base);
338  if (errno || end != value + strlen(value)) return false; // bad parse
339  if (static_cast<uint32>(r) != r) // worked, but number out of range
340  return false;
341  SET_VALUE_AS(uint32, static_cast<uint32>(r));
342  return true;
343  }
344  case FV_INT64: {
345  const int64 r = strto64(value, &end, base);
346  if (errno || end != value + strlen(value)) return false; // bad parse
347  SET_VALUE_AS(int64, r);
348  return true;
349  }
350  case FV_UINT64: {
351  while (*value == ' ') value++;
352  if (*value == '-') return false; // negative number
353  const uint64 r = strtou64(value, &end, base);
354  if (errno || end != value + strlen(value)) return false; // bad parse
355  SET_VALUE_AS(uint64, r);
356  return true;
357  }
358  case FV_DOUBLE: {
359  const double r = strtod(value, &end);
360  if (errno || end != value + strlen(value)) return false; // bad parse
361  SET_VALUE_AS(double, r);
362  return true;
363  }
364  default: {
365  assert(false); // unknown type
366  return false;
367  }
368  }

582 const char* CommandLineFlag::CleanFileName() const {
583  // Compute top-level directory & file that this appears in
584  // search full path backwards.
585  // Stop going backwards at kRootDir; and skip by the first slash.
586  static const char kRootDir[] = ""; // can set this to root directory,
587 
588  if (sizeof(kRootDir)-1 == 0) // no prefix to strip
589  return filename();
590 
591  const char* clean_name = filename() + strlen(filename()) - 1;

793 CommandLineFlag* FlagRegistry::SplitArgumentLocked(const char* arg,
794  string* key,
795  const char** v,
796  string* error_message) {
797  // Find the flag object for this option
798  const char* flag_name;
799  const char* value = strchr(arg, '=');
800  if (value == NULL) {
801  key->assign(arg);
802  *v = NULL;
803  } else {
804  // Strip out the "=value" portion from arg
805  key->assign(arg, value-arg);
806  *v = ++value; // advance past the '='
807  }
808  flag_name = key->c_str();
809 
810  CommandLineFlag* flag = FindFlagLocked(flag_name);
811 
812  if (flag == NULL) {
813  // If we can't find the flag-name, then we should return an error.
814  // The one exception is if 1) the flag-name is 'nox', 2) there
815  // exists a flag named 'x', and 3) 'x' is a boolean flag.
816  // In that case, we want to return flag 'x'.
817  if (!(flag_name[0] == 'n' && flag_name[1] == 'o')) {
818  // flag-name is not 'nox', so we're not in the exception case.
819  *error_message = StringPrintf("%sunknown command line flag '%s'\n",
820  kError, key->c_str());
821  return NULL;
822  }
823  flag = FindFlagLocked(flag_name+2);
824  if (flag == NULL) {
825  // No flag named 'x' exists, so we're not in the exception case.
826  *error_message = StringPrintf("%sunknown command line flag '%s'\n",
827  kError, key->c_str());
828  return NULL;
829  }
830  if (flag->Type() != FlagValue::FV_BOOL) {
831  // 'x' exists but is not boolean, so we're not in the exception case.
832  *error_message = StringPrintf(
833  "%sboolean value (%s) specified for %s command line flag\n",
834  kError, key->c_str(), flag->type_name());
835  return NULL;
836  }
837  // We're in the exception case!
838  // Make up a fake value to replace the "no" we stripped out
839  key->assign(flag_name+2); // the name without the "no"
840  *v = "0";
841  }
842 
843  // Assign a value if this is a boolean flag
844  if (*v == NULL && flag->Type() == FlagValue::FV_BOOL) {
845  *v = "1"; // the --nox case was already handled, so this is the --x case
846  }
847 
848  return flag;
849 }

851 bool TryParseLocked(const CommandLineFlag* flag, FlagValue* flag_value,
852  const char* value, string* msg) {
853  // Use tenative_value, not flag_value, until we know value is valid.
854  FlagValue* tentative_value = flag_value->New();

856  if (msg) {
857  StringAppendF(msg,

859  kError, value,
860  flag->type_name(), flag->name());
861  }
862  delete tentative_value;
863  return false;

870  }
871  delete tentative_value;
872  return false;
873  } else {

875  if (msg) {
876  StringAppendF(msg, "%s set to %s\n",

881  }

884 bool FlagRegistry::SetFlagLocked(CommandLineFlag* flag,
885  const char* value,
886  FlagSettingMode set_mode,
887  string* msg) {
888  flag->UpdateModifiedBit();
889  switch (set_mode) {
890  case SET_FLAGS_VALUE: {
891  // set or modify the flag's value
892  if (!TryParseLocked(flag, flag->current_, value, msg))
893  return false;
894  flag->modified_ = true;
895  break;
896  }
897  case SET_FLAG_IF_DEFAULT: {
898  // set the flag's value, but only if it hasn't been set by someone else
899  if (!flag->modified_) {
900  if (!TryParseLocked(flag, flag->current_, value, msg))
901  return false;
902  flag->modified_ = true;
903  } else {
904  *msg = StringPrintf("%s set to %s",
905  flag->name(), flag->current_value().c_str());
906  }
907  break;
908  }
909  case SET_FLAGS_DEFAULT: {
910  // modify the flag's default-value
911  if (!TryParseLocked(flag, flag->defvalue_, value, msg))
912  return false;
913  if (!flag->modified_) {
914  // Need to set both defvalue *and* current, in this case
915  TryParseLocked(flag, flag->current_, value, NULL);
916  }
917  break;
918  }
919  default: {
920  // unknown set_mode
921  assert(false);
922  return false;
923  }
924  }
925 
926  return true;
927 }

1061 uint32 CommandLineFlagParser::ParseNewCommandLineFlags(int* argc, char*** argv,
1062  bool remove_flags) {
1063  int first_nonopt = *argc; // for non-options moved to the end
1064 
1065  registry_->Lock();
1066  for (int i = 1; i < first_nonopt; i++) {
1067  char* arg = (*argv)[i];
1068 
1069  // Like getopt(), we permute non-option flags to be at the end.
1070  if (arg[0] != '-' || // must be a program argument
1071  (arg[0] == '-' && arg[1] == '\0')) { // "-" is an argument, not a flag
1072  memmove((*argv) + i, (*argv) + i+1, (*argc - (i+1)) * sizeof((*argv)[i]));
1073  (*argv)[*argc-1] = arg; // we go last
1074  first_nonopt--; // we've been pushed onto the stack
1075  i--; // to undo the i++ in the loop
1076  continue;
1077  }
1078 
1079  if (arg[0] == '-') arg++; // allow leading '-'
1080  if (arg[0] == '-') arg++; // or leading '--'
1081 
1082  // -- alone means what it does for GNU: stop options parsing
1083  if (*arg == '\0') {
1084  first_nonopt = i+1;
1085  break;
1086  }
1087 
1088  // Find the flag object for this option
1089  string key;
1090  const char* value;
1091  string error_message;
1092  CommandLineFlag* flag = registry_->SplitArgumentLocked(arg, &key, &value,
1093  &error_message);
1094  if (flag == NULL) {
1095  undefined_names_[key] = ""; // value isn't actually used
1096  error_flags_[key] = error_message;
1097  continue;
1098  }
1099 
1100  if (value == NULL) {
1101  // Boolean options are always assigned a value by SplitArgumentLocked()
1102  assert(flag->Type() != FlagValue::FV_BOOL);
1103  if (i+1 >= first_nonopt) {
1104  // This flag needs a value, but there is nothing available
1105  error_flags_[key] = (string(kError) + "flag '" + (*argv)[i] + "'"
1106  + " is missing its argument");
1107  if (flag->help() && flag->help()[0] > '\001') {
1108  // Be useful in case we have a non-stripped description.
1109  error_flags_[key] += string("; flag description: ") + flag->help();
1110  }
1111  error_flags_[key] += "\n";
1112  break; // we treat this as an unrecoverable error
1113  } else {
1114  value = (*argv)[++i]; // read next arg for value
1115 
1116  // Heuristic to detect the case where someone treats a string arg
1117  // like a bool:
1118  // --my_string_var --foo=bar
1119  // We look for a flag of string type, whose value begins with a
1120  // dash, and where the flag-name and value are separated by a
1121  // space rather than an '='.
1122  // To avoid false positives, we also require the word "true"
1123  // or "false" in the help string. Without this, a valid usage
1124  // "-lat -30.5" would trigger the warning. The common cases we
1125  // want to solve talk about true and false as values.
1126  if (value[0] == '-'
1127  && flag->Type() == FlagValue::FV_STRING
1128  && (strstr(flag->help(), "true")
1129  || strstr(flag->help(), "false"))) {
1130  LOG(WARNING) << "Did you really mean to set flag '"
1131  << flag->name() << "' to the value '"
1132  << value << "'?";
1133  }
1134  }
1135  }
1136 
1137  // TODO(csilvers): only set a flag if we hadn't set it before here
1138  ProcessSingleOptionLocked(flag, value, SET_FLAGS_VALUE);
1139  }
1140  registry_->Unlock();
1141 
1142  if (remove_flags) { // Fix up argc and argv by removing command line flags
1143  (*argv)[first_nonopt-1] = (*argv)[0];
1144  (*argv) += (first_nonopt-1);
1145  (*argc) -= (first_nonopt-1);
1146  first_nonopt = 1; // because we still don't count argv[0]
1147  }
1148 
1149  logging_is_probably_set_up = true; // because we've parsed --logdir, etc.
1150 
1151  return first_nonopt;
1152 }

1307 string CommandLineFlagParser::ProcessOptionsFromStringLocked(
1308  const string& contentdata, FlagSettingMode set_mode) {
1309  string retval;
1310  const char* flagfile_contents = contentdata.c_str();
1311  bool flags_are_relevant = true; // set to false when filenames don't match
1312  bool in_filename_section = false;
1313 
1314  const char* line_end = flagfile_contents;
1315  // We read this file a line at a time.
1316  for (; line_end; flagfile_contents = line_end + 1) {
1317  while (*flagfile_contents && isspace(*flagfile_contents))
1318  ++flagfile_contents;
1319  // Windows uses "\r\n"
1320  line_end = strchr(flagfile_contents, '\r');
1321  if (line_end == NULL)
1322  line_end = strchr(flagfile_contents, '\n');
1323 
1324  size_t len = line_end ? line_end - flagfile_contents
1325  : strlen(flagfile_contents);
1326  string line(flagfile_contents, len);
1327 
1328  // Each line can be one of four things:
1329  // 1) A comment line -- we skip it
1330  // 2) An empty line -- we skip it
1331  // 3) A list of filenames -- starts a new filenames+flags section
1332  // 4) A --flag=value line -- apply if previous filenames match
1333  if (line.empty() || line[0] == '#') {
1334  // comment or empty line; just ignore
1335 
1336  } else if (line[0] == '-') { // flag
1337  in_filename_section = false; // instead, it was a flag-line
1338  if (!flags_are_relevant) // skip this flag; applies to someone else
1339  continue;
1340 
1341  const char* name_and_val = line.c_str() + 1; // skip the leading -
1342  if (*name_and_val == '-')
1343  name_and_val++; // skip second - too
1344  string key;
1345  const char* value;
1346  string error_message;
1347  CommandLineFlag* flag = registry_->SplitArgumentLocked(name_and_val,
1348  &key, &value,
1349  &error_message);
1350  // By API, errors parsing flagfile lines are silently ignored.
1351  if (flag == NULL) {
1352  // "WARNING: flagname '" + key + "' not found\n"
1353  } else if (value == NULL) {
1354  // "WARNING: flagname '" + key + "' missing a value\n"
1355  } else {
1356  retval += ProcessSingleOptionLocked(flag, value, set_mode);
1357  }
1358 
1359  } else { // a filename!
1360  if (!in_filename_section) { // start over: assume filenames don't match
1361  in_filename_section = true;
1362  flags_are_relevant = false;
1363  }
1364 
1365  // Split the line up at spaces into glob-patterns
1366  const char* space = line.c_str(); // just has to be non-NULL
1367  for (const char* word = line.c_str(); *space; word = space+1) {
1368  if (flags_are_relevant) // we can stop as soon as we match
1369  break;
1370  space = strchr(word, ' ');
1371  if (space == NULL)
1372  space = word + strlen(word);
1373  const string glob(word, space - word);
1374  // We try matching both against the full argv0 and basename(argv0)
1375  if (glob == ProgramInvocationName() // small optimization
1376  || glob == ProgramInvocationShortName()
1377 #if defined(HAVE_FNMATCH_H)
1378  || fnmatch(glob.c_str(), ProgramInvocationName(), FNM_PATHNAME) == 0
1379  || fnmatch(glob.c_str(), ProgramInvocationShortName(), FNM_PATHNAME) == 0
1380 #elif defined(HAVE_SHLWAPI_H)
1381  || PathMatchSpec(glob.c_str(), ProgramInvocationName())
1382  || PathMatchSpec(glob.c_str(), ProgramInvocationShortName())
1383 #endif
1384  ) {
1385  flags_are_relevant = true;
1386  }
1387  }
1388  }
1389  }
1390  return retval;
1391 }

1703 // pointers in the 'main' registry, so global FLAG_* vars always

1707 class FlagSaverImpl {
1708  public:
1709  // Constructs an empty FlagSaverImpl object.
1710  explicit FlagSaverImpl(FlagRegistry* main_registry)
1711  : main_registry_(main_registry) { }
1712  ~FlagSaverImpl() {

1714  vector<CommandLineFlag*>::const_iterator it;
1715  for (it = backup_registry_.begin(); it != backup_registry_.end(); ++it)
1716  delete *it;
1717  }

1719  // Saves the flag states from the flag registry into this object.
1720  // It's an error to call this more than once.
1721  // Must be called when the registry mutex is not held.
1722  void SaveFromRegistry() {
1723  FlagRegistryLock frl(main_registry_);
1724  assert(backup_registry_.empty()); // call only once!
1725  for (FlagRegistry::FlagConstIterator it = main_registry_->flags_.begin();
1726  it != main_registry_->flags_.end();
1727  ++it) {
1728  const CommandLineFlag* main = it->second;
1729  // Sets up all the const variables in backup correctly
1730  CommandLineFlag* backup = new CommandLineFlag(
1731  main->name(), main->help(), main->filename(),
1732  main->current_->New(), main->defvalue_->New());
1733  // Sets up all the non-const variables in backup correctly
1734  backup->CopyFrom(*main);
1735  backup_registry_.push_back(backup); // add it to a convenient list
1736  }
1737  }

1740  // assume no flags were added or deleted from the registry since
1741  // the SaveFromRegistry; if they were, that's trouble! Must be
1742  // called when the registry mutex is not held.
1743  void RestoreToRegistry() {
1744  FlagRegistryLock frl(main_registry_);
1745  vector<CommandLineFlag*>::const_iterator it;
1746  for (it = backup_registry_.begin(); it != backup_registry_.end(); ++it) {
1747  CommandLineFlag* main = main_registry_->FindFlagLocked((*it)->name());
1748  if (main != NULL) { // if NULL, flag got deleted from registry(!)
1749  main->CopyFrom(**it);
1750  }
1751  }
1752  }

1756  vector<CommandLineFlag*> backup_registry_;

1759  void operator=(const FlagSaverImpl&);

1961 // ParseCommandLineFlags()
1962 // ParseCommandLineNonHelpFlags()
1963 // HandleCommandLineHelpFlags()
1964 // This is the main function called from main(), to actually
1965 // parse the commandline. It modifies argc and argv as described
1966 // at the top of gflags.h. You can also divide this
1967 // function into two parts, if you want to do work between
1968 // the parsing of the flags and the printing of any help output.
1969 // --------------------------------------------------------------------
1970 
1971 static uint32 ParseCommandLineFlagsInternal(int* argc, char*** argv,
1972  bool remove_flags, bool do_report) {
1973  SetArgv(*argc, const_cast<const char**>(*argv)); // save it for later
1974 
1975  FlagRegistry* const registry = FlagRegistry::GlobalRegistry();
1976  CommandLineFlagParser parser(registry);
1977 
1978  // When we parse the commandline flags, we'll handle --flagfile,
1979  // --tryfromenv, etc. as we see them (since flag-evaluation order
1980  // may be important). But sometimes apps set FLAGS_tryfromenv/etc.
1981  // manually before calling ParseCommandLineFlags. We want to evaluate
1982  // those too, as if they were the first flags on the commandline.
1983  registry->Lock();
1984  parser.ProcessFlagfileLocked(FLAGS_flagfile, SET_FLAGS_VALUE);
1985  // Last arg here indicates whether flag-not-found is a fatal error or not
1986  parser.ProcessFromenvLocked(FLAGS_fromenv, SET_FLAGS_VALUE, true);
1987  parser.ProcessFromenvLocked(FLAGS_tryfromenv, SET_FLAGS_VALUE, false);
1988  registry->Unlock();
1989 
1990  // Now get the flags specified on the commandline
1991  const int r = parser.ParseNewCommandLineFlags(argc, argv, remove_flags);
1992 
1993  if (do_report)
1994  HandleCommandLineHelpFlags(); // may cause us to exit on --help, etc.
1995 
1996  // See if any of the unset flags fail their validation checks
1997  parser.ValidateUnmodifiedFlags();
1998 
1999  if (parser.ReportErrors()) // may cause us to exit on illegal flags
2000  gflags_exitfunc(1);
2001  return r;
2002 }
