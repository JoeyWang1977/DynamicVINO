197 static UNUSED std::vector<std::vector<size_t>> blobToImageOutputArray(InferenceEngine::TBlob<float>::Ptr output,
198  size_t *pWidth, size_t *pHeight,
199  size_t *pChannels) {
200  std::vector<std::vector<size_t>> outArray;
201  size_t W = output->dims().at(0);
202  size_t H = output->dims().at(1);
203  size_t C = output->dims().at(2);
204 
205  // Get classes
206  const float *outData = output->data();
207  for (unsigned h = 0; h < H; h++) {
208  std::vector<size_t> row;
209  for (unsigned w = 0; w < W; w++) {
210  float max_value = outData[h * W + w];
211  size_t index = 0;
212  for (size_t c = 1; c < C; c++) {
213  size_t dataIndex = c * H * W + h * W + w;
214  if (outData[dataIndex] > max_value) {
215  index = c;
216  max_value = outData[dataIndex];
217  }
218  }
219  row.push_back(index);
220  }
221  outArray.push_back(row);
222  }
223 
224  if (pWidth != nullptr) *pWidth = W;
225  if (pHeight != nullptr) *pHeight = H;
226  if (pChannels != nullptr) *pChannels = C;
227 
228  return outArray;
229 }

515  if (y + h >= height) { h = height - y - 1; }
516 
517  size_t shift_first = y*width * 3;
518  size_t shift_second = (y + h)*width * 3;
519  int cls = classes.at(i) % colors.size();
520  for (int i = x; i < x + w; i++) {
521  data[shift_first + i * 3] = colors.at(cls).red();
522  data[shift_first + i * 3 + 1] = colors.at(cls).green();
523  data[shift_first + i * 3 + 2] = colors.at(cls).blue();
524  data[shift_second + i * 3] = colors.at(cls).red();

701  static float ioU(const DetectedObject& detectedObject1_, const DetectedObject& detectedObject2_) {
702  // Add small space to eliminate empty squares
703  float epsilon = 0; // 1e-5f;
704 
705  DetectedObject detectedObject1(detectedObject1_.objectType,
706  (detectedObject1_.xmin - epsilon),
707  (detectedObject1_.ymin - epsilon),
708  (detectedObject1_.xmax- epsilon),
709  (detectedObject1_.ymax- epsilon), detectedObject1_.prob);
710  DetectedObject detectedObject2(detectedObject2_.objectType,
711  (detectedObject2_.xmin + epsilon),
712  (detectedObject2_.ymin + epsilon),
713  (detectedObject2_.xmax),
714  (detectedObject2_.ymax), detectedObject2_.prob);
715 
716  if (detectedObject1.objectType != detectedObject2.objectType) {
717  // objects are different, so the result is 0
718  return 0.0f;
719  }
720 
721  if (detectedObject1.xmax < detectedObject1.xmin) return 0.0;
722  if (detectedObject1.ymax < detectedObject1.ymin) return 0.0;
723  if (detectedObject2.xmax < detectedObject2.xmin) return 0.0;
724  if (detectedObject2.ymax < detectedObject2.ymin) return 0.0;
725 
726 
727  float xmin = (std::max)(detectedObject1.xmin, detectedObject2.xmin);
728  float ymin = (std::max)(detectedObject1.ymin, detectedObject2.ymin);
729  float xmax = (std::min)(detectedObject1.xmax, detectedObject2.xmax);
730  float ymax = (std::min)(detectedObject1.ymax, detectedObject2.ymax);
731 
732  // Caffe adds 1 to every length if the box isn't normalized. So do we...
733  float addendum;
734  if (xmax > 1 || ymax > 1)
735  addendum = 1;
736  else
737  addendum = 0;
738 
739  // intersection
740  float intr;
741  if ((xmax >= xmin) && (ymax >= ymin)) {
742  intr = (addendum + xmax - xmin) * (addendum + ymax - ymin);
743  } else {
744  intr = 0.0f;
745  }
746 
747  // union
748  float square1 = (addendum + detectedObject1.xmax - detectedObject1.xmin) * (addendum + detectedObject1.ymax - detectedObject1.ymin);
749  float square2 = (addendum + detectedObject2.xmax - detectedObject2.xmin) * (addendum + detectedObject2.ymax - detectedObject2.ymin);
750 
751  float unn = square1 + square2 - intr;
752 
753  return static_cast<float>(intr) / unn;
754  }
