96 void ValidatedFaceDetectionNetwork::checkLayerProperty(
97  const InferenceEngine::CNNNetReader::Ptr &net_reader) {
98  slog::info << "Checking Face Detection outputs" << slog::endl;
99  InferenceEngine::OutputsDataMap
100  output_info_map(net_reader->getNetwork().getOutputsInfo());
101  InferenceEngine::DataPtr &output_data_ptr = output_info_map.begin()->second;
102  output_ = output_info_map.begin()->first;
103  const InferenceEngine::CNNLayerPtr
104  output_layer = net_reader->getNetwork().getLayerByName(
105  output_.c_str());
106  //output layer should be DetectionOutput type
107  if (output_layer->type != "DetectionOutput") {
108  throw std::logic_error(
109  "Face Detection network output layer(" + output_layer->name +
110  ") should be DetectionOutput, but was " + output_layer->type);
111  }
112  //output layer should have attribute called num_classes
113  if (output_layer->params.find("num_classes") ==
114  output_layer->params.end()) {
115  throw std::logic_error("Face Detection network output layer (" +
116  output_+
117  ") should have num_classes integer attribute");
118  }
119  //class number should be equal to size of label vector
120  //if network has default "background" class, fake is used
121  const int num_classes = output_layer->GetParamAsInt("num_classes");
122  if (getLabels().size() != num_classes) {
123  if (getLabels().size() == (num_classes - 1)) {
124  getLabels().insert(getLabels().begin(), "fake");
125  } else {
126  getLabels().clear();
127  }
128  }
129  //last dimension of output layer should be 7
130  const InferenceEngine::SizeVector output_dims
131  = output_data_ptr->getTensorDesc().getDims();
132  max_proposal_count_ = static_cast<int>(output_dims[2]);
133  slog::info << "max proposal count is: " << max_proposal_count_ <<slog::endl;
134  object_size_= static_cast<int>(output_dims[3]);
135  if (object_size_ != 7) {
136  throw std::logic_error(
137  "Face Detection network output layer should have 7 as a last dimension"
138  );
139  }
140  if (output_dims.size() != 4) {
141  throw std::logic_error(
142  "Face Detection network output dimensions not compatible shoulld be 4, "
143  "but was " + std::to_string(output_dims.size()));
144  }
145 }

277 void ValidatedHeadPoseNetwork::checkLayerProperty(
278  const InferenceEngine::CNNNetReader::Ptr &net_reader) {
279  slog::info << "Checking Head Pose network outputs" << slog::endl;
280  InferenceEngine::OutputsDataMap outputInfo(
281  net_reader->getNetwork().getOutputsInfo());
282  std::map<std::string, bool> layerNames = {
283  {output_angle_r_, false},
284  {output_angle_p_, false},
285  {output_angle_y_, false}
286  };
287 
288  for (auto &&output : outputInfo) {
289  InferenceEngine::CNNLayerPtr layer =
290  output.second->getCreatorLayer().lock();
291  if (layerNames.find(layer->name) == layerNames.end()) {
292  throw std::logic_error(
293  "Head Pose network output layer unknown: " + layer->name
294  + ", should be " +
295  output_angle_r_ + " or " + output_angle_p_ + " or "
296  + output_angle_y_);
297  }
298  if (layer->type != "FullyConnected") {
299  throw std::logic_error("Head Pose network output layer (" + layer->name
300  + ") has invalid type: " +
301  layer->type + ", should be FullyConnected");
302  }
303  auto fc = dynamic_cast<InferenceEngine::FullyConnectedLayer *>(layer.get());
304  if (fc->_out_num != 1) {
305  throw std::logic_error("Head Pose network output layer (" + layer->name
306  + ") has invalid out-size=" +
307  std::to_string(fc->_out_num) + ", should be 1");
308  }
309  layerNames[layer->name] = true;
310  }
311 }
